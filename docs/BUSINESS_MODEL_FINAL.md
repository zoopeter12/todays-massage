```sql
-- 본인인증 관련 컬럼 추가
ALTER TABLE profiles ADD COLUMN ci TEXT UNIQUE;
ALTER TABLE profiles ADD COLUMN di TEXT UNIQUE;
ALTER TABLE profiles ADD COLUMN real_name TEXT;
ALTER TABLE profiles ADD COLUMN gender TEXT CHECK (gender IN ('male', 'female'));
ALTER TABLE profiles ADD COLUMN birth_date DATE;
ALTER TABLE profiles ADD COLUMN verified_at TIMESTAMPTZ;
ALTER TABLE profiles ADD COLUMN credit_score INTEGER DEFAULT 100;

-- 인덱스 생성
CREATE INDEX idx_profiles_di ON profiles(di);
CREATE INDEX idx_profiles_credit_score ON profiles(credit_score);
```

#### blacklist 테이블 생성

```sql
CREATE TABLE blacklist (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  di TEXT UNIQUE NOT NULL,
  reason TEXT,
  blocked_at TIMESTAMPTZ DEFAULT NOW(),
  blocked_by UUID REFERENCES profiles(id),
  original_user_id UUID,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 인덱스 생성
CREATE INDEX idx_blacklist_di ON blacklist(di);
CREATE INDEX idx_blacklist_blocked_at ON blacklist(blocked_at);
```

#### 신용점수 변동 이력 테이블

```sql
CREATE TABLE credit_score_history (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES profiles(id) NOT NULL,
  previous_score INTEGER NOT NULL,
  new_score INTEGER NOT NULL,
  change_amount INTEGER NOT NULL,
  reason TEXT NOT NULL,
  reservation_id UUID REFERENCES reservations(id),
  created_at TIMESTAMPTZ DEFAULT NOW(),
  created_by UUID REFERENCES profiles(id)
);

CREATE INDEX idx_credit_history_user ON credit_score_history(user_id);
```

## 6. 경쟁 우위 분석

### 6.1 경쟁사 대비 장점

| 항목 | 경쟁사 | 오늘의마사지 | 차이점 |
|------|--------|-------------|--------|
| **월 이용료** | 마통 33만원 | 0원 | 연 400만원 절감 |
| **입점비** | 힐리 입점비 있음 | 0원 | 진입장벽 제거 |
| **전화번호** | 대부분 공개 | 완전 비공개 | 직거래 방지 |
| **본인인증** | 약함 (휴대폰만) | 다날 PASS | 문제고객 영구차단 |
| **수수료** | 10~15% | 9.9% | 업계 최저 수준 |

### 6.2 매장 입장 가치 제안

- **무료 입점**: 리스크 없이 온라인 예약 채널 확보
- **낮은 수수료**: 실질 수수료 6.5%로 마진 보존
- **직거래 방지**: 전화번호 비공개로 플랫폼 우회 차단
- **문제고객 차단**: 신용점수 시스템으로 노쇼/악성 고객 필터링
- **월 2회 정산**: 안정적인 현금 흐름

### 6.3 고객 입장 가치 제안

- **신규가입 혜택**: 10,000원 쿠폰으로 첫 경험 유도
- **출석/룰렛 보상**: 지속적인 앱 방문 동기
- **안심 예약**: 본인인증 매장만 등록되어 신뢰도 향상
- **채팅 문의**: 전화 부담 없이 편리한 문의
